getContent			
	zwraca string podany jako parametr
	
getLower
	zwraca ilość małych liter ze stringa
	
getUpper
	zwraca ilość dużych liter w stringu podanym jako parametr
	
getNumber
	zwraca ilość liczb w stringu podanym jako parametr

getSpecial
	zwraca ilość znaków specjalnych w stringu podanym jako parametr
